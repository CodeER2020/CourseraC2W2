Sub test()

Dim ws As Worksheet
Dim last As Long
Dim i As Long

For Each ws In Worksheets(Array("Test"))
    ws.Activate
    last = Cells(Rows.Count, "A").End(xlUp).Row 'Delete rows
    For i = last To 2 Step -1
        If Cells(i, "F").Value = "ELIMINAR" Then
            Cells(i, "F").EntireRow.Delete
        End If
     Next i
    Next ws
    
End Sub
